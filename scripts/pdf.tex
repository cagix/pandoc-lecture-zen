%%-- Coloured boxes
\setlength{\FrameSep}{8pt}                              %% padding
\setlength{\FrameRule}{2pt}                             %% width of left bar

\newcommand{\AlertTitleFont}{\normalsize\bfseries}      %% alternativ: \small\bfseries

\newenvironment{AlertBox}[3]{%
  \par\addvspace{.8\baselineskip}%
  \def\FrameCommand{%
    {\color{#2}\vrule width \FrameRule}\hspace{\FrameSep}%
    \colorbox{#3}%
  }%
  \MakeFramed{\advance\hsize-\width \FrameRestore}%
  \noindent{\AlertTitleFont #1}\par\medskip
}{%
  \endMakeFramed%
  \par\addvspace{.4\baselineskip}%
}
